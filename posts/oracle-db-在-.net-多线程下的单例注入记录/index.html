using Oracle.ManagedDataAccess.Client;
using System;
using System.Data;

public class OracleConnectionManager : IOracleConnectionManager
{
    private readonly string _connectionString;

    public OracleConnectionManager(string connectionString)
    {
        _connectionString = connectionString;
    }

    public void Execute(Action<IDbConnection> action)
    {
        using (var connection = new OracleConnection(_connectionString))
        {
            connection.Open ();
            action (connection);
        }
    }

    public T Execute<T>(Func<IDbConnection, T> func)
    {
        using (var connection = new OracleConnection(_connectionString))
        {
            connection.Open ();
            return func(connection);
        }
    }

    public void Dispose ()
    {
        // Dispose if necessary
    }
}

public interface IOracleConnectionManager : IDisposable
{
    void Execute(Action<IDbConnection> action);
    T Execute<T>(Func<IDbConnection, T> func);
}
